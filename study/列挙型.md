# 列挙型

javaの列挙型は、特定の値の集合を定義するための特別なクラスです

型として扱えるため、型安全性が向上します。
数値や文字列とは違い、他の値を代入することができないため、誤った値を使用するリスクが減ります。

列挙型の例
```java
// 曜日を表す列挙型
public enum Day {
    MONDAY,
    TUESDAY,
    WEDNESDAY,
    THURSDAY,
    FRIDAY,
    SATURDAY,
    SUNDAY
}

// 定数に情報や振る舞いを持たせる
public enum Level {
    LOW(1),    // 各列挙子の後ろに()でパラメータを渡す
    MEDIUM(2),
    HIGH(3);

    private final int levelCode;      // フィールドの定義
    public int getLevelCode() {      // メソッドの定義（ゲッター）
        return this.levelCode;
    }
}

// 定数に個別の振る舞いを持たせる
public enum Operation {
    ADD {
        @Override
        public double apply(double x, double y) {
            return x + y;
        }
    },
    SUBTRACT {
        @Override
        public double apply(double x, double y) {
            return x - y;
        }
    },
    MULTIPLY {
        @Override
        public double apply(double x, double y) {
            return x * y;
        }
    },
    DIVIDE {
        @Override
        public double apply(double x, double y) {
            return x / y;
        }
    };

    public abstract double apply(double x, double y);
}


// 状態を表す列挙型
public enum TrafficLight {
    RED, GREEN, YELLOW;
    public TrafficLight next() {
        switch(this) {
            case RED:    return GREEN;
            case GREEN:  return YELLOW;
            case YELLOW: return RED;
        }
    }
}

```
